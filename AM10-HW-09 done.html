# Define Euler's method
def euler(Nx, xmin, xmax, uf0):
    #Diff eqn of form: d(uf)/dx=ff(x,uf) 
    #Nx: Number of points to be used for integration
    #xmin, xmax: Initial and final values of the independent variable (x)
    #uf0: Initial value of the dependent variable at x=xmin
    
    #Assume the function f(u,x) is of form:
    
    #ff(x[n], uf[n])
    
    #For any function f(u,x) on the RHS, you need to provide (define it outside of this euler() function, in a separate cell)
    #a python function. For example, for the eigenfunction problem:
    #Define x
    x = np.linspace(xmin, xmax, Nx)
    #Calculate dx
    dx = x[1] - x[0]
    #Set up the array for ux
    uf = np.zeros(Nx)
    #Define the initial condition
    uf[0] = uf0
    
    #We need to write another function ff(x[n],uf[n])
    #for the integrand. We will do it later when we need to
    #solve the equation for particular cases.
    for n in range(Nx - 1):
        #Fill in the code for uf[n+1]
        #--------------------------------------
        uf[n+1] = uf[n] + dx * ff(x[n], uf[n])
        #--------------------------------------
    return uf
# Define modified Euler's method
def mod_euler(Nx, xmin, xmax, uf0):
    #Diff eqn of form: d(uf)/dx=ff(x,uf)
    #Nx: Number of points to be used for integration
    #xmin, xmax: Initial and final values of the independent variable (x)
    #uf0: Initial value of the dependent variable at x=xmin
     
    #Define x
    x = np.linspace(xmin, xmax, Nx)
    #Determine dx
    dx = x[1] - x[0]
    #Set up the array for ux and initialize it to zero
    uf = np.zeros(Nx)
    #Specify the boundary (initial) condition at x=xmin
    uf[0] = uf0

    for n in range(Nx - 1):
        # uf[n+1] is of the form: uf[n+1] = uf[n] + 0.5 * dx * (ff1 + ff2)
        #Write in the code for ff1 and ff2
        #--------------------------------------
        ff1 = ff(x[n], uf[n])
        ff2 = ff(x[n+1], uf[n] + dx * ff(x[n], uf[n]))
        #--------------------------------------------
        uf[n+1] = uf[n] + 0.5 * dx * (ff1 + ff2)
    return uf

Problem 2¶

Expand the SIR epidemic model to include an additional compartment, that of the Exposed (E) population.¶

In many contagious diseases there is a significant incubation period during which individuals have been infected but are not yet infectious themselves. This delay between the acquisition of infection and the infectious state can be included within the SIR model by adding an exposed population (compartment), E, and letting the infected (but not yet infectious) individuals move from the susceptible population (S) to E and from E to I (Infectious). This expanded model is called SEIR.

Assume a closed population with no births or deaths (that is, the overall population, N, remains constant). The differential equations describing this model are:

$$ \frac{dS}{dt} = - \beta I \frac{S}{N} $$$$ \frac{dE}{dt} = \beta I \frac{S}{N} - \sigma E$$$$ \frac{dI}{dt} = \sigma E - \gamma I $$$$ \frac{dR}{dt} = \gamma I $$

where $1 / \sigma$ is the average incubation period (that is, the average time of remaining in compartment E), and all other constants and variables have their usual meaning.

a) Solve the system of differential equations of the SEIR model for incubation period $1/\sigma = 2$ days, and for incubation period $1/\sigma = 5$ days. The numerical values of the other parameters can be obtained from the examples solved in the Lecture or the Lab.

b) Plot the time evolution (for 200 days) of the epidemic for all populations (S, E, I, R) in one graph, and comment on the results.

# Total population (N) and number of days (Ndays) of epidemic duration
N = 1000
Ndays=200

# Contact rate (beta), mean recovery rate (gamma,in 1/days), and incubation period (1/sigma)
beta = 0.3
gamma = 1.0/14
sigma=1/2

# Generating the time array:
t = np.linspace(0, Ndays, Ndays)
dt=t[1]-t[0]

# Initializing the S,E, I, R arrays
S=np.zeros(Ndays)
E=np.zeros(Ndays)
I=np.zeros(Ndays)
R=np.zeros(Ndays)

# start with one infected individual
S[0]=N
E[0]=0
I[0]=1
R[0]=0

# Solving the system of differential equations
for n in range(Ndays-1):
    S[n+1]=S[n]+dt*(-beta*S[n]*I[n]/N)
    E[n+1]=E[n]+dt*(beta*I[n]*S[n]/N-sigma*E[n])
    I[n+1]=I[n]+dt*(sigma*E[n] - gamma*I[n])
    R[n+1]=R[n]+dt*(gamma*I[n])


# Plotting the results (disease trajectory)
fig=plt.figure(figsize=(8,6))
plt.plot(t,S,'bo-',ms=4,label='Susceptible')
plt.plot(t,E,'ko-',ms=4,label='Exposed')
plt.plot(t,I,'ro-',ms=4,label='Infectious')
plt.plot(t,R,'go-',ms=4,label='Recovered')

plt.xlabel('Time (days)')
plt.ylabel('Number')
plt.ylim(0,01.05*N)
plt.legend(loc='center right')
plt.title("SEIR Model with Average Incubation Period of 2 Days")
plt.show()

#define sigmoid 
def sigmoid(a,t0,t):
    return 1/(1+np.exp(-a*(t-t0)))

def fit_and_plot_func(func, x, y, title): #define fit and plot function
    # first find the coefficients
    popt, pcov = curve_fit(func, x, y) 
    num_vars = len(popt)
    print(f'Number of variables used = {num_vars}')
    
    # Plot the polynomial 
    ffit=func(x,*popt) 
    
    # calculate the rmse
    Nd=np.shape(x)
    ovrNd=(1.0/np.sqrt(Nd))
    rmsefit=ovrNd*np.sqrt(np.sum((ffit-y)**2))

    print(f'RMSE = {rmsefit}')
    plt.figure(figsize=(8,5))
    plt.title(title)
    plt.scatter(x,y, facecolor = 'blue', alpha = 0.6, label='data points')
    
    plt.plot(x,ffit,'g', label='fitting function') 
    plt.xlabel(r'$x$')
    plt.ylabel(r'$f(x)$')
    plt.legend()
    plt.show();

#define fit 
def daily_fit(t,t1,dt,N2):
    t01 = t1
    t02 = t01 + dt
    return N2*(sigmoid(alpha1,t01,t)-sigmoid(alpha2,t02,t))
