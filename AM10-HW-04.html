def gauss1D(x,mu,sigma):
    peak=1.0/(np.sqrt(2.0*np.pi)*sigma)
    return peak*np.exp(-(x-mu)**2/(2.0*sigma**2))

def epic(t,a,b):
    x = (a+b)*np.cos(t)-(b*np.cos((a/b+1)*t))
    y = (a+b)*np.sin(t)-(b*np.sin((a/b+1)*t))
    plt.plot(x,y)
    plt.title('Epicycloid')
    plt.xlabel('X (t)')
    plt.ylabel('Y (t)')
    plt.axis("equal")
